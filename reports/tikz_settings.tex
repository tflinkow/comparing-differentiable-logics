\usepackage{tikz}
\usepackage{pgfplots}
\usepackage{pgfplotstable}
\usepgfplotslibrary{colormaps, colorbrewer, groupplots}

\usepackage{amsmath}

\pgfplotsset{
  compat=1.18,
  cycle list/Set1,
  % colorbrewer cycle lists don't typically define markers, so here we go:
  % https://tikz.dev/pgfplots/libs-colorbrewer#autosec-8544
  cycle multiindex* list={
        mark list*\nextlist
        Set1\nextlist
    },
  results/.style={
    width=5.5cm,
    ytick distance=0.2,
    xtick={0,10,...,50},
    tick label style={font=\scriptsize},
    legend cell align={left},
    legend style={
      at={($(0,0)+(1cm,1cm)$)},
      legend columns=3,
      anchor=center,
      align=center,
      font=\scriptsize
    },
    legend image post style={mark indices={}},
    enlargelimits=true,
    xlabel=Epoch,
    ylabel=Accuracy,
    every axis plot post/.append style={
      thick,
      table/col sep=comma,
      table/x=Epoch,
    },
  },
  group/results/.style={
    group style={
      group size=2 by 1, 
      ylabels at=edge left,
      x descriptions at=edge bottom,
      horizontal sep=0.25cm,
    }, 
    results,
  },
}
